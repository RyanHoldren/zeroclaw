1//! Shell-based tool derived from a skill's `[[tools]]` section.
2//!
3//! Each `SkillTool` with `kind = "shell"` or `kind = "script"` is converted
4//! into a `SkillShellTool` that implements the `Tool` trait. The tool name is
5//! prefixed with the skill name (e.g. `my_skill.run_lint`) to avoid collisions
6//! with built-in tools.
7
8use crate::security::SecurityPolicy;
9use async_trait::async_trait;
10use std::collections::HashMap;
11use std::sync::Arc;
12use std::time::Duration;
13use zeroclaw_api::tool::{Tool, ToolResult};
14
15/// Default execution time for a skill shell command (seconds).
16const SKILL_SHELL_TIMEOUT_SECS: u64 = 60;
17/// Maximum output size in bytes (1 MB).
18const MAX_OUTPUT_BYTES: usize = 1_048_576;
19
20/// A tool derived from a skill's `[[tools]]` section that executes shell commands.
21pub struct SkillShellTool {
22    tool_name: String,
23    tool_description: String,
24    command_template: String,
25    args: HashMap<String, String>,
26    timeout_secs: u64,
27    security: Arc<SecurityPolicy>,
28}
29
30impl SkillShellTool {
31    /// Create a new skill shell tool.
32    ///
33    /// The tool name is prefixed with the skill name (`skill_name.tool_name`)
34    /// to prevent collisions with built-in tools.
35    pub fn new(
36        skill_name: &str,
37        tool: &crate::skills::SkillTool,
38        security: Arc<SecurityPolicy>,
39    ) -> Self {
40        Self {
41            tool_name: format!("{}.{}", skill_name, tool.name),
42            tool_description: tool.description.clone(),
43            command_template: tool.command.clone(),
44            args: tool.args.clone(),
45            timeout_secs: tool.timeout_secs.unwrap_or(SKILL_SHELL_TIMEOUT_SECS).max(1),
46            security,
47        }
48    }
49
50    fn build_parameters_schema(&self) -> serde_json::Value {
51        let mut properties = serde_json::Map::new();
52        let mut required = Vec::new();
53
54        for (name, description) in &self.args {
55            properties.insert(
56                name.clone(),
57                serde_json::json!({
58                    "type": "string",
59                    "description": description
60                }),
61            );
62            required.push(serde_json::Value::String(name.clone()));
63        }
64
65        serde_json::json!({
66            "type": "object",
67            "properties": properties,
68            "required": required
69        })
70    }
71
72    /// Substitute `{{arg_name}}` placeholders in the command template with
73    /// the provided argument values. Unknown placeholders are left as-is.
74    fn substitute_args(&self, args: &serde_json::Value) -> String {
75        let mut command = self.command_template.clone();
76        if let Some(obj) = args.as_object() {
77            for (key, value) in obj {
78                let placeholder = format!("{{{{{}}}}}", key);
79                let replacement = value.as_str().unwrap_or_default();
80                command = command.replace(&placeholder, replacement);
81            }
82        }
83        command
84    }
85}

87#[async_trait]
88impl Tool for SkillShellTool {
89    fn name(&self) -> &str {
90        &self.tool_name
91    }
92
93    fn description(&self) -> &str {
94        &self.tool_description
95    }
96
97    fn parameters_schema(&self) -> serde_json::Value {
98        self.build_parameters_schema()
99    }
100
101    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
102        let command = self.substitute_args(&args);
103
104        // Rate limiting is applied by the RateLimitedTool wrapper at
105        // registration time (see zeroclaw-runtime::tools::mod). The
106        // PathGuardedTool wrapper cannot inspect the substituted command
107        // built by substitute_args, so the forbidden_path_argument check
108        // below remains tool-local.
109
110        // Security validation — always requires explicit approval (approved=true)
111        // since skill tools are user-defined and should be treated as medium-risk.
112        match self.security.validate_command_execution(&command, true) {
113            Ok(_) => {}
114            Err(reason) => {
115                return Ok(ToolResult {
116                    success: false,
117                    output: String::new(),
118                    error: Some(reason),
119                });
120            }
121        }
122
123        if let Some(path) = self.security.forbidden_path_argument(&command) {
124            return Ok(ToolResult {
125                success: false,
126                output: String::new(),
127                error: Some(format!("Path blocked by security policy: {path}")),
128            });
129        }
130
131        // Build and execute the command
132        let mut cmd = tokio::process::Command::new("sh");
133        cmd.arg("-c").arg(&command);
134        cmd.current_dir(&self.security.workspace_dir);
135        cmd.env_clear();
136        // Ensure the child is killed when the timeout future is dropped.
137        cmd.kill_on_drop(true);
138
139        // Only pass safe environment variables
140        for var in &[
141            "PATH", "HOME", "TERM", "LANG", "LC_ALL", "USER", "SHELL", "TMPDIR",
142        ] {
143            if let Ok(val) = std::env::var(var) {
144                cmd.env(var, val);
145            }
146        }
147
148        let result =
149            tokio::time::timeout(Duration::from_secs(self.timeout_secs), cmd.output()).await;
150
151        match result {
152            Ok(Ok(output)) => {
153                let mut stdout = String::from_utf8_lossy(&output.stdout).to_string();
154                let mut stderr = String::from_utf8_lossy(&output.stderr).to_string();
155
156                if stdout.len() > MAX_OUTPUT_BYTES {
157                    let mut b = MAX_OUTPUT_BYTES.min(stdout.len());
158                    while b > 0 && !stdout.is_char_boundary(b) {
159                        b -= 1;
160                    }
161                    stdout.truncate(b);
162                    stdout.push_str("\n... [output truncated at 1MB]");
163                }
164                if stderr.len() > MAX_OUTPUT_BYTES {
165                    let mut b = MAX_OUTPUT_BYTES.min(stderr.len());
166                    while b > 0 && !stderr.is_char_boundary(b) {
167                        b -= 1;
168                    }
169                    stderr.truncate(b);
170                    stderr.push_str("\n... [stderr truncated at 1MB]");
171                }
172
173                Ok(ToolResult {
174                    success: output.status.success(),
175                    output: stdout,
176                    error: if stderr.is_empty() {
177                        None
178                    } else {
179                        Some(stderr)
180                    },
181                })
182            }
183            Ok(Err(e)) => Ok(ToolResult {
184                success: false,
185                output: String::new(),
186                error: Some(format!("Failed to execute command: {e}")),
187            }),
188            Err(_) => Ok(ToolResult {
189                success: false,
190                output: String::new(),
191                error: Some(format!(
192                    "Command timed out after {}s and was killed",
193                    self.timeout_secs
194                )),
195            }),
196        }
197    }
198}

373    #[tokio::test]
374    async fn skill_shell_tool_timeout_kills_process() {
375        let marker = std::env::temp_dir().join("zeroclaw_kill_marker_test");
376        let _ = std::fs::remove_file(&marker);
377
378        let st = SkillTool {
379            name: "slow".to_string(),
380            description: "".to_string(),
381            kind: "shell".to_string(),
382            command: format!("sleep 3 && touch {}", marker.display()),
383            args: HashMap::new(),
384            timeout_secs: Some(1),
385        };
386        let tool = SkillShellTool::new("test", &st, test_security());
387        let result = tool.execute(serde_json::json!({})).await.unwrap();
388        assert!(!result.success);
389
390        // Wait past the sleep duration — if the shell wasn't killed the touch will run.
391        tokio::time::sleep(Duration::from_secs(4)).await;
392        assert!(!marker.exists(), "shell process was not killed on timeout");
393    }
